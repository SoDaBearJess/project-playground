// brewing logic, simplified

I love writing code that reads like a recipe — clear, composable, and easy for the next person to follow. Here’s a tiny taste: a typed brew controller that balances ratios just like I balance front and back end.

More on GitHub ↗
brew.ts 
// The golden ratio: 1g coffee : 16g water
type Brew = { coffee: number; style: "pourover" | "espresso" };

const brew = ({ coffee, style }: Brew) => {
  const ratio = style === "espresso" ? 2 : 16;
  const water = coffee * ratio;

  return {
    water,
    strength: water < 200 ? "bold" : "smooth",
    pour() {
      return `Pour ${water}g over ${coffee}g ☕`;
    },
  };
};

// ship it ➜
console.log(brew({ coffee: 22, style: "pourover" }).pour());
